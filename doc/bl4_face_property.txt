00000000 SSSSSSSS 00D0020B GGW0R00V

V = Is Visible
R = Is Road
W = Is Wall
% = ? seen with walls
G = Graphic level (higher means less textured if graphics low)
B = Black is transparent
2 = Two faced texture, no culling
D = Dural effect
S = Slipperyness
  = Animated Texture Data
  = Reflect zones 0-7

00000000 00000000 00000000 01100000 (0x60) = Invisible Wall
00000000 00000000 00000000 01100001 (0x61) = Visible Wall
00000000 00000000 00000000 00001001 (0x9) = Visible Road
